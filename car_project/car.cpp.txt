#include <iostream>
#include <string>

using namespace std;

struct Car {
    string brand;
    string model;
    int year;
    int horsepower;

    void printInfo() {
        cout << "Марка: " << brand << endl;
        cout << "Модель: " << model << endl;
        cout << "Год выпуска: " << year << endl;
        cout << "Мощность: " << horsepower << " HP" << endl;
    }
};

double powerToKW(int hp) {
    return hp * 0.735;
}

int main() {
    Car myCar = { "Toyota", "Camry", 2020, 200 };
    myCar.printInfo();
    cout << "Мощность в кВт: " << powerToKW(myCar.horsepower) << endl;
    return 0;
}
